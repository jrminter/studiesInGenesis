% a note "by the way"
\newenvironment{btw}			
	{%enter
		\par\vspace{4pt}
        \begin{quote}
		\footnotesize\textbf{Note: }%
		}
	{%exit
    \end{quote}
		}

% tasks: per section
\newcounter{task}			
\newenvironment{task}	 % 	a question	
	{%enter
		\stepcounter{task}
		\setlength{\parindent}{0pt}
		\par\vspace{4pt}
		\rule{\textwidth}{1pt} \\
		\textbf{\textsc{Task} \arabic{task}} : %
		}
	{%exit
		\hspace*{\fill}$\bullet$
		\par
		}

% the ambitious reader can try...
\newenvironment{challenge}
	{%enter
		\par\vspace{4pt}
		\textbf{\textcolor{ForestGreen}{Challenge}: }%
		}
	{%exit
		}

% questions: whole doc
\newcounter{question}			
\newenvironment{question}				% a question
	{%enter
		\stepcounter{question}
        \hypertarget{Q\arabic{question}}{}
		\setlength{\parindent}{0pt}
		\par\vspace{4pt}
		\rule{\textwidth}{1pt} \\
		\textbf{\textsc{Q}\arabic{question}} : %
		\slshape
		}
	{%exit
		\hspace*{\fill}
        \hyperlink{A\arabic{question}}{Jump to A\arabic{question}}
        $\bullet$
		\par\vspace{4pt}
		}
